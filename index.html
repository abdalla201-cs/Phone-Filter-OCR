#!/bin/bash echo "" echo " ==========================================" echo " Phone Number Processor - Starting..." echo " ==========================================" echo "" PORT=8000 FILE="phone-processor-gemini.html" URL="http://localhost:$PORT/$FILE" open_browser() { if command -v open &>/dev/null; then # macOS open "$URL" elif command -v xdg-open &>/dev/null; then # Linux xdg-open "$URL" fi } # Try Python 3 if command -v python3 &>/dev/null; then echo " [OK] Python3 found — server at $URL" sleep 1 && open_browser & python3 -m http.server $PORT exit 0 fi # Try Python 2 if command -v python &>/dev/null; then echo " [OK] Python found — server at $URL" sleep 1 && open_browser & python -m SimpleHTTPServer $PORT exit 0 fi # Try npx serve if command -v npx &>/dev/null; then echo " [OK] Node.js found — server at http://localhost:3000/$FILE" sleep 2 && open "http://localhost:3000/$FILE" & npx serve -p 3000 . exit 0 fi echo " [ERROR] Python or Node.js not found." echo "" echo " Install one of:" echo " Python: https://www.python.org/downloads/" echo " Node.js: https://nodejs.org/" echo ""
